    ```typescript
    export type BreakthroughCheck =
      | { allowed: true }
      | { allowed: false; reason: 'max_realm' }
      | { allowed: false; reason: 'insufficient_tuvi'; required: number; current: bigint };

    export type BreakthroughResult =
      | { outcome: 'success'; newRealmId: number }
      | { outcome: 'fail'; penaltyAmount: bigint };

    export function canAttemptBreakthrough(char: { realmId: number; tuVi: bigint }): BreakthroughCheck {
      if (char.realmId >= 41) return { allowed: false, reason: 'max_realm' };
      const currentTier = REALM_CONFIG[char.realmId];
      const required = currentTier.tuViRequired;
      if (char.tuVi < required) return { allowed: false, reason: 'insufficient_tuvi', required, current: char.tuVi };
      return { allowed: true };
    }

    export function rollBreakthrough(char: { realmId: number; tuVi: bigint }): BreakthroughResult {
      const currentTier = REALM_CONFIG[char.realmId];
      const isBoundary = currentTier.isMajorBoundary;
      const failChance = isBoundary ? currentTier.failureChance : 0;
      const roll = Math.random();
      if (roll < failChance) {
        // Failure: lose 50% of excess tu vi above entry threshold
        const excess = char.tuVi - BigInt(currentTier.entryThreshold);
        const penalty = excess > 0n ? excess / 2n : 0n;
        return { outcome: 'fail', penaltyAmount: penalty };
      }
      return { outcome: 'success', newRealmId: char.realmId + 1 };
    }

    // DB write functions (use db from client.ts):
    export async function applyBreakthroughSuccess(characterId: number, newRealmId: number): Promise<void>;
    export async function applyBreakthroughFailure(characterId: number, penaltyAmount: bigint): Promise<void>;
    // Success: UPDATE characters SET realm_id = newRealmId WHERE id = characterId
    // Failure: UPDATE characters SET tu_vi = tu_vi - penaltyAmount WHERE id = characterId AND tu_vi - penaltyAmount >= 0
    // (Guard: tu_vi never drops below 0 even if penalty calculation had a bug)
    ```
